291  uint8_t chipId = caerSpikeEventGetChipID(event);
292  uint8_t coreId = caerSpikeEventGetSourceCoreID(event);
293  uint32_t neuronId = caerSpikeEventGetNeuronID(event);

296  bool addColumn = (coreId & 0x01);
