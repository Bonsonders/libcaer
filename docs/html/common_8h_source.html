390 static inline caerEventPacketHeader caerGenericEventPacketGrow(caerEventPacketHeader packet, int32_t newEventCapacity) {
391  if (packet == NULL || newEventCapacity == 0) {
392  return (NULL);
393  }
394 
395  int32_t oldEventCapacity = caerEventPacketHeaderGetEventCapacity(packet);
396 
397  if (newEventCapacity <= oldEventCapacity) {
398  caerLog(CAER_LOG_CRITICAL, "Generic Event Packet",
399  "Called caerGenericEventPacketGrow() with a new capacity value (%" PRIi32 ") that is equal or smaller than the old one (%" PRIi32 "). "
400  "Only strictly growing an event packet is supported!", newEventCapacity, oldEventCapacity);
401  return (NULL);
402  }
403 
404  int32_t eventSize = caerEventPacketHeaderGetEventSize(packet);
405  size_t newEventPacketSize = CAER_EVENT_PACKET_HEADER_SIZE + (size_t) (newEventCapacity * eventSize);
406 
407  // Grow memory used to hold events.
408  packet = (caerEventPacketHeader) realloc(packet, newEventPacketSize);
409  if (packet == NULL) {
410  caerLog(CAER_LOG_CRITICAL, "Generic Event Packet",
411  "Failed to reallocate %zu bytes of memory for growing Event Packet of capacity %"
412  PRIi32 " to new capacity of %" PRIi32 ". Error: %d.", newEventPacketSize, oldEventCapacity,
413  newEventCapacity, errno);
414  return (NULL);
415  }
416 
417  // Zero out new event memory (all events invalid).
418  memset(((uint8_t *) packet) + CAER_EVENT_PACKET_HEADER_SIZE + (oldEventCapacity * eventSize), 0,
419  (size_t) ((newEventCapacity - oldEventCapacity) * eventSize));
420 
421  // Update header fields.
422  caerEventPacketHeaderSetEventCapacity(packet, newEventCapacity);
423 
424  return (packet);
425 }

441 static inline caerEventPacketHeader caerGenericEventPacketAppend(caerEventPacketHeader packet,
442  caerEventPacketHeader appendPacket) {
443  if (packet == NULL) {
444  return (NULL);
445  }
446 
447  // Support appending nothing, the result is the unmodified input.
448  if (appendPacket == NULL) {
449  return (packet);
450  }
451 
452  // Check that the two packets are of the same type and size, and have the same TSOverflow epoch.
453  if ((caerEventPacketHeaderGetEventType(packet) != caerEventPacketHeaderGetEventType(appendPacket))
454  || (caerEventPacketHeaderGetEventSize(packet) != caerEventPacketHeaderGetEventSize(appendPacket))
455  || (caerEventPacketHeaderGetEventTSOverflow(packet) != caerEventPacketHeaderGetEventTSOverflow(appendPacket))) {
456  return (NULL);
457  }
458 
459  int32_t packetEventValid = caerEventPacketHeaderGetEventValid(packet);
460  int32_t packetEventNumber = caerEventPacketHeaderGetEventNumber(packet);
461  int32_t packetEventCapacity = caerEventPacketHeaderGetEventCapacity(packet);
462 
463  int32_t appendPacketEventValid = caerEventPacketHeaderGetEventValid(appendPacket);
464  int32_t appendPacketEventNumber = caerEventPacketHeaderGetEventNumber(appendPacket);
465  int32_t appendPacketEventCapacity = caerEventPacketHeaderGetEventCapacity(appendPacket);
466 
467  int32_t eventSize = caerEventPacketHeaderGetEventSize(packet); // Is the same! Checked above.
468  size_t newEventPacketSize = CAER_EVENT_PACKET_HEADER_SIZE
469  + (size_t) ((packetEventCapacity + appendPacketEventCapacity) * eventSize);
470 
471  // Grow memory used to hold events.
472  packet = (caerEventPacketHeader) realloc(packet, newEventPacketSize);
473  if (packet == NULL) {
474  caerLog(CAER_LOG_CRITICAL, "Generic Event Packet",
475  "Failed to reallocate %zu bytes of memory for appending Event Packet of capacity %"
476  PRIi32 " to Event Packet of capacity %" PRIi32 ". Error: %d.", newEventPacketSize,
477  appendPacketEventCapacity, packetEventCapacity, errno);
478  return (NULL);
479  }
480 
481  // Copy appendPacket event memory at start of free space in packet.
482  memcpy(((uint8_t *) packet) + CAER_EVENT_PACKET_HEADER_SIZE + (packetEventNumber * eventSize),
483  ((uint8_t *) appendPacket) + CAER_EVENT_PACKET_HEADER_SIZE, (size_t) (appendPacketEventNumber * eventSize));
484 
485  // Zero out remaining event memory (all events invalid).
486  memset(
487  ((uint8_t *) packet) + CAER_EVENT_PACKET_HEADER_SIZE
488  + ((packetEventNumber + appendPacketEventNumber) * eventSize), 0,
489  (size_t) (((packetEventCapacity + appendPacketEventCapacity) - (packetEventNumber + appendPacketEventNumber))
490  * eventSize));
491 
492  // Update header fields.
493  caerEventPacketHeaderSetEventValid(packet, (packetEventValid + appendPacketEventValid));
494  caerEventPacketHeaderSetEventNumber(packet, (packetEventNumber + appendPacketEventNumber));
495  caerEventPacketHeaderSetEventCapacity(packet, (packetEventCapacity + appendPacketEventCapacity));
496 
497  return (packet);
498 }

616 static inline void *caerCopyEventPacket(void *eventPacket) {
617  // Handle empty event packets.
618  if (eventPacket == NULL) {
619  return (NULL);
620  }
621 
622  // Calculate needed memory for new event packet.
623  caerEventPacketHeader header = (caerEventPacketHeader) eventPacket;
624  int32_t eventSize = caerEventPacketHeaderGetEventSize(header);
625  int32_t eventNumber = caerEventPacketHeaderGetEventNumber(header);
626  int32_t eventCapacity = caerEventPacketHeaderGetEventCapacity(header);
627  size_t packetMem = CAER_EVENT_PACKET_HEADER_SIZE + (size_t) (eventSize * eventCapacity);
628  size_t dataMem = CAER_EVENT_PACKET_HEADER_SIZE + (size_t) (eventSize * eventNumber);
629 
630  // Allocate memory for new event packet.
631  caerEventPacketHeader eventPacketCopy = (caerEventPacketHeader) malloc(packetMem);
632  if (eventPacketCopy == NULL) {
633  // Failed to allocate memory.
634  return (NULL);
635  }
636 
637  // Copy the data over.
638  memcpy(eventPacketCopy, eventPacket, dataMem);
639 
640  // Zero out the rest of the packet.
641  memset(((uint8_t *) eventPacketCopy) + dataMem, 0, packetMem - dataMem);
642 
643  return (eventPacketCopy);
644 }

749 static inline void caerCleanEventPacket(void *eventPacket) {
750  // Handle empty event packets.
751  if (eventPacket == NULL) {
752  return;
753  }
754 
755  // Calculate needed memory for new event packet.
756  caerEventPacketHeader header = (caerEventPacketHeader) eventPacket;
757  int32_t eventSize = caerEventPacketHeaderGetEventSize(header);
758  int32_t eventValid = caerEventPacketHeaderGetEventValid(header);
759  int32_t eventNumber = caerEventPacketHeaderGetEventNumber(header);
760  int32_t eventCapacity = caerEventPacketHeaderGetEventCapacity(header);
761 
762  // If we have no invalid events, we're already done.
763  if (eventValid == eventNumber) {
764  return;
765  }
766 
767  // Move all valid events close together. Must check every event for validity!
768  size_t offset = CAER_EVENT_PACKET_HEADER_SIZE;
769 
770  CAER_ITERATOR_VALID_START(header, void *)
771  void *dest = ((uint8_t *) header) + offset;
772 
773  if (dest != caerIteratorElement) {
774  memcpy(dest, caerIteratorElement, (size_t) eventSize);
775  offset += (size_t) eventSize;
776  }
777  }
778 
779  // Reset remaining memory, up to capacity, to zero (all events invalid).
780  memset(((uint8_t *) header) + offset, 0, (size_t) ((eventCapacity - eventValid) * eventSize));
781 
782  // Event capacity remains unchanged, event number shrunk to event valid number.
783  caerEventPacketHeaderSetEventNumber(header, eventValid);
784 }
