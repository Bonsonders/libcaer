498 static inline void caerEventPacketClean(void *eventPacket) {
499  // Handle empty event packets.
500  if (eventPacket == NULL) {
501  return;
502  }
503 
504  // Calculate needed memory for new event packet.
505  caerEventPacketHeader header = (caerEventPacketHeader) eventPacket;
506  int32_t eventValid = caerEventPacketHeaderGetEventValid(header);
507  int32_t eventNumber = caerEventPacketHeaderGetEventNumber(header);
508 
509  // If we have no invalid events, we're already done.
510  if (eventValid == eventNumber) {
511  return;
512  }
513 
514  int32_t eventSize = caerEventPacketHeaderGetEventSize(header);
515  int32_t eventCapacity = caerEventPacketHeaderGetEventCapacity(header);
516 
517  // Move all valid events close together. Must check every event for validity!
518  size_t offset = CAER_EVENT_PACKET_HEADER_SIZE;
519 
520  CAER_ITERATOR_VALID_START(header, const void *)
521  void *dest = ((uint8_t *) header) + offset;
522 
523  if (dest != caerIteratorElement) {
524  memcpy(dest, caerIteratorElement, (size_t) eventSize);
525  offset += (size_t) eventSize;
526  }
527  }
528 
529  // Reset remaining memory, up to capacity, to zero (all events invalid).
530  memset(((uint8_t *) header) + offset, 0, (size_t) ((eventCapacity - eventValid) * eventSize));
531 
532  // Event capacity remains unchanged, event number shrunk to event valid number.
533  caerEventPacketHeaderSetEventNumber(header, eventValid);
534 }

554 static inline caerEventPacketHeader caerEventPacketResize(caerEventPacketHeader packet, int32_t newEventCapacity) {
555  if (packet == NULL || newEventCapacity == 0) {
556  return (NULL);
557  }
558 
559  // Always clean for consistency with shrink case (side-effects guarantee).
560  caerEventPacketClean(packet);
561 
562  int32_t oldEventCapacity = caerEventPacketHeaderGetEventCapacity(packet);
563 
564  if (oldEventCapacity == newEventCapacity) {
565  // Nothing to do in this case.
566  return (packet);
567  }
568 
569  int32_t eventSize = caerEventPacketHeaderGetEventSize(packet);
570  size_t newEventPacketSize = CAER_EVENT_PACKET_HEADER_SIZE + (size_t) (newEventCapacity * eventSize);
571 
572  // Reallocate memory used to hold events.
573  packet = (caerEventPacketHeader) realloc(packet, newEventPacketSize);
574  if (packet == NULL) {
575  caerLog(CAER_LOG_CRITICAL, "Event Packet",
576  "Failed to reallocate %zu bytes of memory for resizing Event Packet of capacity %"
577  PRIi32 " to new capacity of %" PRIi32 ". Error: %d.", newEventPacketSize, oldEventCapacity,
578  newEventCapacity, errno);
579  return (NULL);
580  }
581 
582  if (newEventCapacity > oldEventCapacity) {
583  // Capacity increased: we simply zero out the newly added events.
584  size_t oldEventPacketSize = CAER_EVENT_PACKET_HEADER_SIZE + (size_t) (oldEventCapacity * eventSize);
585 
586  memset(((uint8_t *) packet) + oldEventPacketSize, 0,
587  (size_t) ((newEventCapacity - oldEventCapacity) * eventSize));
588  }
589  else {
590  // Capacity decreased: the events were cleaned, so eventValid == eventNumber.
591  // They also are all together in memory. Thus we can simply keep the current
592  // eventValid/eventNumber counts if the capacity is still bigger or equal to
593  // them, or, if new capacity is smaller, we reset them to that value.
594  int32_t oldEventNumber = caerEventPacketHeaderGetEventNumber(packet);
595 
596  if (newEventCapacity < oldEventNumber) {
597  caerEventPacketHeaderSetEventValid(packet, newEventCapacity);
598  caerEventPacketHeaderSetEventNumber(packet, newEventCapacity);
599  }
600  }
601 
602  // Update capacity header field.
603  caerEventPacketHeaderSetEventCapacity(packet, newEventCapacity);
604 
605  return (packet);
606 }

675 static inline caerEventPacketHeader caerEventPacketAppend(caerEventPacketHeader packet,
676  caerEventPacketHeader appendPacket) {
677  if (packet == NULL) {
678  return (NULL);
679  }
680 
681  // Support appending nothing, the result is the unmodified input.
682  if (appendPacket == NULL) {
683  return (packet);
684  }
685 
686  // Check that the two packets are of the same type and size, and have the same TSOverflow epoch.
687  if ((caerEventPacketHeaderGetEventType(packet) != caerEventPacketHeaderGetEventType(appendPacket))
688  || (caerEventPacketHeaderGetEventSize(packet) != caerEventPacketHeaderGetEventSize(appendPacket))
689  || (caerEventPacketHeaderGetEventTSOverflow(packet) != caerEventPacketHeaderGetEventTSOverflow(appendPacket))) {
690  return (NULL);
691  }
692 
693  int32_t packetEventValid = caerEventPacketHeaderGetEventValid(packet);
694  int32_t packetEventNumber = caerEventPacketHeaderGetEventNumber(packet);
695  int32_t packetEventCapacity = caerEventPacketHeaderGetEventCapacity(packet);
696 
697  int32_t appendPacketEventValid = caerEventPacketHeaderGetEventValid(appendPacket);
698  int32_t appendPacketEventNumber = caerEventPacketHeaderGetEventNumber(appendPacket);
699  int32_t appendPacketEventCapacity = caerEventPacketHeaderGetEventCapacity(appendPacket);
700 
701  int32_t eventSize = caerEventPacketHeaderGetEventSize(packet); // Is the same! Checked above.
702  size_t newEventPacketSize = CAER_EVENT_PACKET_HEADER_SIZE
703  + (size_t) ((packetEventCapacity + appendPacketEventCapacity) * eventSize);
704 
705  // Grow memory used to hold events.
706  packet = (caerEventPacketHeader) realloc(packet, newEventPacketSize);
707  if (packet == NULL) {
708  caerLog(CAER_LOG_CRITICAL, "Generic Event Packet",
709  "Failed to reallocate %zu bytes of memory for appending Event Packet of capacity %"
710  PRIi32 " to Event Packet of capacity %" PRIi32 ". Error: %d.", newEventPacketSize,
711  appendPacketEventCapacity, packetEventCapacity, errno);
712  return (NULL);
713  }
714 
715  // Copy appendPacket event memory at start of free space in packet.
716  memcpy(((uint8_t *) packet) + CAER_EVENT_PACKET_HEADER_SIZE + (packetEventNumber * eventSize),
717  ((uint8_t *) appendPacket) + CAER_EVENT_PACKET_HEADER_SIZE, (size_t) (appendPacketEventNumber * eventSize));
718 
719  // Zero out remaining event memory (all events invalid).
720  memset(
721  ((uint8_t *) packet) + CAER_EVENT_PACKET_HEADER_SIZE
722  + ((packetEventNumber + appendPacketEventNumber) * eventSize), 0,
723  (size_t) (((packetEventCapacity + appendPacketEventCapacity) - (packetEventNumber + appendPacketEventNumber))
724  * eventSize));
725 
726  // Update header fields.
727  caerEventPacketHeaderSetEventValid(packet, (packetEventValid + appendPacketEventValid));
728  caerEventPacketHeaderSetEventNumber(packet, (packetEventNumber + appendPacketEventNumber));
729  caerEventPacketHeaderSetEventCapacity(packet, (packetEventCapacity + appendPacketEventCapacity));
730 
731  return (packet);
732 }
