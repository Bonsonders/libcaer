490 static inline void caerEventPacketClean(void *eventPacket) {
491  // Handle empty event packets.
492  if (eventPacket == NULL) {
493  return;
494  }
495 
496  // Calculate needed memory for new event packet.
497  caerEventPacketHeader header = (caerEventPacketHeader) eventPacket;
498  int32_t eventValid = caerEventPacketHeaderGetEventValid(header);
499  int32_t eventNumber = caerEventPacketHeaderGetEventNumber(header);
500 
501  // If we have no invalid events, we're already done.
502  if (eventValid == eventNumber) {
503  return;
504  }
505 
506  int32_t eventSize = caerEventPacketHeaderGetEventSize(header);
507  int32_t eventCapacity = caerEventPacketHeaderGetEventCapacity(header);
508 
509  // Move all valid events close together. Must check every event for validity!
510  size_t offset = CAER_EVENT_PACKET_HEADER_SIZE;
511 
512  CAER_ITERATOR_VALID_START(header, const void *)
513  void *dest = ((uint8_t *) header) + offset;
514 
515  if (dest != caerIteratorElement) {
516  memcpy(dest, caerIteratorElement, (size_t) eventSize);
517  offset += (size_t) eventSize;
518  }
519  }
520 
521  // Reset remaining memory, up to capacity, to zero (all events invalid).
522  memset(((uint8_t *) header) + offset, 0, (size_t) ((eventCapacity - eventValid) * eventSize));
523 
524  // Event capacity remains unchanged, event number shrunk to event valid number.
525  caerEventPacketHeaderSetEventNumber(header, eventValid);
526 }

546 static inline caerEventPacketHeader caerEventPacketResize(caerEventPacketHeader packet, int32_t newEventCapacity) {
547  if (packet == NULL || newEventCapacity == 0) {
548  return (NULL);
549  }
550 
551  // Always clean for consistency with shrink case (side-effects guarantee).
552  caerEventPacketClean(packet);
553 
554  int32_t oldEventCapacity = caerEventPacketHeaderGetEventCapacity(packet);
555 
556  if (oldEventCapacity == newEventCapacity) {
557  // Nothing to do in this case.
558  return (packet);
559  }
560 
561  int32_t eventSize = caerEventPacketHeaderGetEventSize(packet);
562  size_t newEventPacketSize = CAER_EVENT_PACKET_HEADER_SIZE + (size_t) (newEventCapacity * eventSize);
563 
564  // Reallocate memory used to hold events.
565  packet = (caerEventPacketHeader) realloc(packet, newEventPacketSize);
566  if (packet == NULL) {
567  caerLog(CAER_LOG_CRITICAL, "Event Packet",
568  "Failed to reallocate %zu bytes of memory for resizing Event Packet of capacity %"
569  PRIi32 " to new capacity of %" PRIi32 ". Error: %d.", newEventPacketSize, oldEventCapacity,
570  newEventCapacity, errno);
571  return (NULL);
572  }
573 
574  if (newEventCapacity > oldEventCapacity) {
575  // Capacity increased: we simply zero out the newly added events.
576  size_t oldEventPacketSize = CAER_EVENT_PACKET_HEADER_SIZE + (size_t) (oldEventCapacity * eventSize);
577 
578  memset(((uint8_t *) packet) + oldEventPacketSize, 0,
579  (size_t) ((newEventCapacity - oldEventCapacity) * eventSize));
580  }
581  else {
582  // Capacity decreased: the events were cleaned, so eventValid == eventNumber.
583  // They also are all together in memory. Thus we can simply keep the current
584  // eventValid/eventNumber counts if the capacity is still bigger or equal to
585  // them, or, if new capacity is smaller, we reset them to that value.
586  int32_t oldEventNumber = caerEventPacketHeaderGetEventNumber(packet);
587 
588  if (newEventCapacity < oldEventNumber) {
589  caerEventPacketHeaderSetEventValid(packet, newEventCapacity);
590  caerEventPacketHeaderSetEventNumber(packet, newEventCapacity);
591  }
592  }
593 
594  // Update capacity header field.
595  caerEventPacketHeaderSetEventCapacity(packet, newEventCapacity);
596 
597  return (packet);
598 }

667 static inline caerEventPacketHeader caerEventPacketAppend(caerEventPacketHeader packet,
668  caerEventPacketHeader appendPacket) {
669  if (packet == NULL) {
670  return (NULL);
671  }
672 
673  // Support appending nothing, the result is the unmodified input.
674  if (appendPacket == NULL) {
675  return (packet);
676  }
677 
678  // Check that the two packets are of the same type and size, and have the same TSOverflow epoch.
679  if ((caerEventPacketHeaderGetEventType(packet) != caerEventPacketHeaderGetEventType(appendPacket))
680  || (caerEventPacketHeaderGetEventSize(packet) != caerEventPacketHeaderGetEventSize(appendPacket))
681  || (caerEventPacketHeaderGetEventTSOverflow(packet) != caerEventPacketHeaderGetEventTSOverflow(appendPacket))) {
682  return (NULL);
683  }
684 
685  int32_t packetEventValid = caerEventPacketHeaderGetEventValid(packet);
686  int32_t packetEventNumber = caerEventPacketHeaderGetEventNumber(packet);
687  int32_t packetEventCapacity = caerEventPacketHeaderGetEventCapacity(packet);
688 
689  int32_t appendPacketEventValid = caerEventPacketHeaderGetEventValid(appendPacket);
690  int32_t appendPacketEventNumber = caerEventPacketHeaderGetEventNumber(appendPacket);
691  int32_t appendPacketEventCapacity = caerEventPacketHeaderGetEventCapacity(appendPacket);
692 
693  int32_t eventSize = caerEventPacketHeaderGetEventSize(packet); // Is the same! Checked above.
694  size_t newEventPacketSize = CAER_EVENT_PACKET_HEADER_SIZE
695  + (size_t) ((packetEventCapacity + appendPacketEventCapacity) * eventSize);
696 
697  // Grow memory used to hold events.
698  packet = (caerEventPacketHeader) realloc(packet, newEventPacketSize);
699  if (packet == NULL) {
700  caerLog(CAER_LOG_CRITICAL, "Generic Event Packet",
701  "Failed to reallocate %zu bytes of memory for appending Event Packet of capacity %"
702  PRIi32 " to Event Packet of capacity %" PRIi32 ". Error: %d.", newEventPacketSize,
703  appendPacketEventCapacity, packetEventCapacity, errno);
704  return (NULL);
705  }
706 
707  // Copy appendPacket event memory at start of free space in packet.
708  memcpy(((uint8_t *) packet) + CAER_EVENT_PACKET_HEADER_SIZE + (packetEventNumber * eventSize),
709  ((uint8_t *) appendPacket) + CAER_EVENT_PACKET_HEADER_SIZE, (size_t) (appendPacketEventNumber * eventSize));
710 
711  // Zero out remaining event memory (all events invalid).
712  memset(
713  ((uint8_t *) packet) + CAER_EVENT_PACKET_HEADER_SIZE
714  + ((packetEventNumber + appendPacketEventNumber) * eventSize), 0,
715  (size_t) (((packetEventCapacity + appendPacketEventCapacity) - (packetEventNumber + appendPacketEventNumber))
716  * eventSize));
717 
718  // Update header fields.
719  caerEventPacketHeaderSetEventValid(packet, (packetEventValid + appendPacketEventValid));
720  caerEventPacketHeaderSetEventNumber(packet, (packetEventNumber + appendPacketEventNumber));
721  caerEventPacketHeaderSetEventCapacity(packet, (packetEventCapacity + appendPacketEventCapacity));
722 
723  return (packet);
724 }
